    def nextSymbol
      skipWhitespace

      case @kind
        when :letter
          return processLetter
        when :digit
          return processNumber
        when :special
          return processSpecial
        when :eof
          # end of input
          return LexerSymbol.new(:eoi)
        else
          raise Exception
      end
    end
