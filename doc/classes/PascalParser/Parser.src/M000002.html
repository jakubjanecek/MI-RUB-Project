    def parse
      @lexer.open

      readSymbol

      begin
        # start symbol
        program()
        puts "Parsed successfully."
      rescue Exception => err
        puts "Error at line #{@lexer.lineNumber}, character #{@lexer.characterPosition}:\n\t#{err.to_s}"
      ensure
        @lexer.close
      end
    end
